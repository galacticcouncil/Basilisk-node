Examples
use wasmparser::AliasSectionReader;
let data: &[u8] = &[0x01, 0x02, 0x00, 0x03, b'f', b'o', b'o'];
let mut reader = AliasSectionReader::new(data, 0).unwrap();
for _ in 0..reader.get_count() {
    let alias = reader.read().expect("alias");
    println!("Alias: {:?}", alias);
}
